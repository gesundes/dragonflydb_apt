Quick Setup

Add this repository to your system:

# Add GPG key (install into apt keyrings)
sudo install -d -m 0755 /etc/apt/keyrings
curl -fsSL https://gesundes.github.io/dragonflydb_apt/pubkey.gpg | sudo gpg --dearmor -o /etc/apt/keyrings/dragonflydb.gpg
sudo chmod 0644 /etc/apt/keyrings/dragonflydb.gpg

# Add repository (use signed-by via keyring)
echo "deb [signed-by=/etc/apt/keyrings/dragonflydb.gpg] https://gesundes.github.io/dragonflydb_apt stable main" | sudo tee /etc/apt/sources.list.d/dragonflydb.list >/dev/null

# Recommend enabling By-Hash for stronger metadata integrity
echo 'Acquire::By-Hash "true";' | sudo tee /etc/apt/apt.conf.d/99-by-hash >/dev/null

# Update package list
sudo apt update

# Install DragonflyDB
sudo apt install dragonfly
